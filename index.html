How to use

#include <stdio.h>

#include "include/ccollection.h"

int main()
{
    vector_t * vec = vector_new(sizeof(int));

    int value = 100, i;
    vector_push_back(vec, &value);
    vector_push_back(vec, &value);
    vector_push_back(vec, &value);

    for (i = 0; i < vector_get_size(vec); i++)
    {
        vector_at(vec, i, &value);
        printf("%d ", value);
    }
    printf("\n");
    vector_destroy(vec);
}
